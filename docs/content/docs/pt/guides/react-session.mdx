## SessionProvider

Envolva seu app com `SessionProvider` para disponibilizar a sessão em todos os componentes cliente.

```tsx title="src/pages/_app.tsx"
import type { AppProps } from 'vinext/app';
import { SessionProvider } from 'vinextauth/react';

export default function App({ Component, pageProps }: AppProps) {
  return (
    <SessionProvider session={pageProps.session}>
      <Component {...pageProps} />
    </SessionProvider>
  );
}
```

### Props

| Prop                   | Tipo              | Padrão        | Descrição                                                |
| ---------------------- | ----------------- | ------------- | -------------------------------------------------------- |
| `session`              | `Session \| null` | `undefined`   | Sessão do servidor para evitar fetch inicial no cliente. |
| `basePath`             | `string`          | `"/api/auth"` | Sobrescreve o caminho base do auth.                      |
| `refetchInterval`      | `number`          | —             | Re-busca a sessão a cada N segundos.                     |
| `refetchOnWindowFocus` | `boolean`         | `true`        | Re-busca quando a aba volta ao foco.                     |

## useSession

```tsx
'use client';
import { useSession } from 'vinextauth/react';

export function MenuUsuario() {
  const { data: session, status } = useSession();

  if (status === 'loading') return <span>Carregando...</span>;
  if (status === 'unauthenticated') return <a href="/api/auth/signin">Entrar</a>;

  return <span>Olá, {session.user.name}</span>;
}
```

### Valor retornado

| Campo    | Tipo                                                | Descrição                                      |
| -------- | --------------------------------------------------- | ---------------------------------------------- |
| `data`   | `Session \| null`                                   | Sessão atual ou `null` quando não autenticado. |
| `status` | `"loading" \| "authenticated" \| "unauthenticated"` | Estado de carregamento da sessão.              |
| `update` | `(data?) => Promise<Session \| null>`               | Atualiza ou recarrega a sessão.                |

## signIn

Redireciona para a página de login ou diretamente para um provider OAuth:

```ts
import { signIn } from 'vinextauth/react';

signIn(); // Página de login
signIn('google'); // Direto para o Google
signIn('github', { callbackUrl: '/dashboard' }); // Com redirect customizado
```

## signOut

Faz POST no endpoint de logout e redireciona:

```ts
import { signOut } from 'vinextauth/react';

await signOut(); // Logout e redirect para a origem
await signOut({ callbackUrl: '/login' }); // Redirect customizado
```
